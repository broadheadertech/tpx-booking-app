### 2. Send Time Optimizer

```javascript
const getOptimalSendTime = (campaigns) => {
  const hourlyOpenRates = {}

  campaigns.forEach(c => {
    const hour = new Date(c.sent_at).getHours()
    if (!hourlyOpenRates[hour]) hourlyOpenRates[hour] = []
    hourlyOpenRates[hour].push(c.open_count / c.sent_count)
  })

  const bestHour = Object.entries(hourlyOpenRates)
    .map(([hour, rates]) => ({
      hour,
      avgRate: rates.reduce((a,b) => a+b, 0) / rates.length
    }))
    .sort((a,b) => b.avgRate - a.avgRate)[0]

  return bestHour
}
```

### 3. Churn Risk Scoring

```javascript
const calculateChurnRisk = (customer) => {
  const daysSinceVisit = (Date.now() - customer.lastVisit) / (1000*60*60*24)
  const avgVisitInterval = customer.avgDaysBetweenVisits || 30

  if (daysSinceVisit < avgVisitInterval) return { risk: 'low', score: 20 }
  if (daysSinceVisit < avgVisitInterval * 1.5) return { risk: 'medium', score: 50 }
  if (daysSinceVisit < avgVisitInterval * 2) return { risk: 'high', score: 75 }
  return { risk: 'critical', score: 95, action: 'Send win-back NOW!' }
}
```

const calculateRFM = (customer) => {
